### 模板

```
void backtracking(参数) {
    if (终止条件) {
        存放结果;
        return;
    }
    for (选择：本层集合中元素（树中节点孩子的数量就是集合的大小）) {
        处理节点;
        backtracking(路径，选择列表); // 递归
        回溯，撤销处理结果
    }
}
```

### 八皇后问题
```c++
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> ans;
        vector<std::string> cases(n, std::string(n, '.'));
        backtrace(ans, cases, 0, n);
        return ans;
    }

    void backtrace(vector<vector<string>> &ans, vector<string> &cases, int row, int n){
        if(row == n){
            ans.push_back(cases);
            return;
        }

        for(int j = 0; j < n; j++){
            if(isValid(row, j, cases, n)){
                cases[row][j] = 'Q';

                backtrace(ans, cases, row+1, n);

                cases[row][j] = '.';

            } 
        }
    }
    
    bool isValid(int row, int col, vector<string> v, int n){
        for(int i=0; i<row; i++){
            if(v[i][col] == 'Q')
                return false;
        }

        for(int i=row-1, j=col+1; i>=0 && j<n; i--, j++){
            if(v[i][j] == 'Q')
                return false;
        }

        for(int i=row-1, j=col-1; i>=0 && j>=0; i--, j--){
            if(v[i][j] == 'Q')
                return false;
        }
        return true;
    }
};
```
